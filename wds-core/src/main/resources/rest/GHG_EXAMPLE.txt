http://localhost:8080/wds/api?out=html
							 &db=faostat
							 &select=D.Year[Year],A.AreaNameE[Country],I.ItemNameE[Item],E.ElementNameE[Element],D.Value
							 &from=Data[D],Area[A],Item[I],Element[E]
							 &where=A.AreaCode(2),
							 		D.DomainCode('QC'),
							 		JOIN(D.AreaCode:A.AreaCode),
							 		JOIN(D.ElementCode:E.ElementCode),							 							 
							 		JOIN(D.ItemCode:I.ItemCode),
							 		D.Value(0)
							 &groupby=D.Year,A.AreaNameE,I.ItemNameE,E.ElementNameE,D.Value
							 &orderby=D.Year,A.AreaNameE,I.ItemNameE,E.ElementNameE

							 &db=faostat
							 &select=D.Year[Year],A.AreaNameE[Country],I.ItemNameE[Item],E.ElementNameE[Element]
							 &from=Data[D],Area[A],Item[I],Element[E]
							 &where=A.AreaCode(79),
							 		I.ItemCode(1034),
							 		D.GroupCode('Q'),
							 		D.DomainCode('QA'),
							 		JOIN(D.AreaCode:A.AreaCode),
							 		JOIN(D.ElementCode:E.ElementCode),							 							 
							 		JOIN(D.ItemCode:I.ItemCode)
							 &groupby=D.Year,A.AreaNameE,I.ItemNameE,E.ElementNameE
							 &orderby=D.Year,A.AreaNameE,I.ItemNameE,E.ElementNameE
